import { Injectable } from '@angular/core';
import { HttpHeaders, HttpClient } from '@angular/common/http';
import { Observable } from 'rxjs';
import { environment } from 'src/environments/environment';




import { NlpRequest, NlpResponse } from '../models/nlp';
import { VectorizerRequest, VectorizerResponse } from '../models/vectorizer';
import { CodebookRequest, CodebookResponse } from '../models/codebook';
import { EmbeddingRequest, EmbeddingResponse } from '../models/embedding';
import { AllStepParamsRequest, AllStepsParametersResponse } from '../models/allStepParams';
import { AllTasksReponse } from '../models/alltask';


/**
 * Servicio que contiene las peticiones al api de SYN.
 */
@Injectable({
  providedIn: 'root'
})
export class SynApiService {

  constructor(private http: HttpClient) { }


  allTasks(training: string): Observable<AllTasksReponse> {
    return this.http.get<AllTasksReponse>(environment.apiUri + '/tasks/' + training)
  }

  getAllParams(allStepParamsRequest: AllStepParamsRequest): Observable<AllStepsParametersResponse> {
    return this.http.get<AllStepsParametersResponse>(environment.apiUri + '/tasks/' + allStepParamsRequest.id_task + '/steps/' + allStepParamsRequest.parametro_peticion + '/parameters')
  }
  /**
   * Petici√≥n que convierte lo introducido en {@Link InputComponent} a la respuesta usada en {@Link NlpComponent}
   * @param nlpRequest cuerpo del request
   */
  doNlp(nlpRequest: NlpRequest): Observable<NlpResponse> {
    return this.http.post<NlpResponse>(environment.apiDomain + '/nlp', nlpRequest.transform());
  }

  /**
   * Petici√≥n que convierte lo devuelto por doNlp la respuesta usada en {@Link EmbeddingComponent}
   * @param embeddingRequest cuerpo del request
   */
  doEmbedding(embeddingRequest: EmbeddingRequest): Observable<EmbeddingResponse> {
    return this.http.post<EmbeddingResponse>(environment.apiDomain + '/embeddings' + '/?corpus=openOffice', embeddingRequest.transform());
  }

  /**
   * Petici√≥n que convierte lo devuelto por doEmbedding la respuesta usada en {@Link CodebookComponent}
   * @param codebookRequest cuerpo del request
   */
  doCodebook(codebookRequest: CodebookRequest): Observable<CodebookResponse> {
    return this.http.post<CodebookResponse>(environment.apiDomain + '/codebooks' + '/?corpus=openOffice', codebookRequest.transform());
  }

  /**
   * Petici√≥n que convierte lo devuelto por doCodebook la respuesta usada en {@Link VectorizerComponent}
   * @param codebookRequest cuerpo del request
   */
  doVectorizer(vectorizerRequest: VectorizerRequest): Observable<VectorizerResponse> {
    return this.http.post<VectorizerResponse>(environment.apiDomain + '/vectorizer' + '/?corpus=openOffice', vectorizerRequest.transform());
  }



}
