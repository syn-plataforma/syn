import { Component, Input, OnInit } from '@angular/core';
import { FormBuilder, FormGroup } from '@angular/forms';
import { BehaviorSubject, Observable } from 'rxjs';
import { delay } from 'rxjs/operators';

/**
 * Componente contenido en {@Link TourComponent}.
 *
 * Este componente representa
 */
@Component({
  selector: 'app-codebook',
  templateUrl: './codebook.component.html',
  styleUrls: ['./codebook.component.scss']
})
export class CodebookComponent implements OnInit {

  @Input() data;
  step = 'codebooks';
  frmStepFour: FormGroup;
  frmStepFour$: Observable<FormGroup>;

  private myFrmStepFour$ = new BehaviorSubject<FormGroup>(null);
  myFrmStepFourListener$: Observable<FormGroup> = this.myFrmStepFour$.asObservable();

  myFrmStepFour(form: FormGroup) {
    this.myFrmStepFour$.next(form);
  }

  constructor(private formBuilder: FormBuilder) {
    this.frmStepFour = this.formBuilder.group({
      address: ['']
    });
  }

  ngOnInit(): void {
    this.myFrmStepFour(this.frmStepFour);

    this.frmStepFour$ = this.myFrmStepFourListener$.pipe(
      delay(0)
    );
  }

}
