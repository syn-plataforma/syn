@Component({
  selector: 'app-index',
  templateUrl: './index.component.html',
  styleUrls: ['./index.component.scss']
})
export class IndexComponent implements OnInit {

  indexToTourModel: IndexToTourModel;
  
  treatments = treatments;
  listaTask: Task[] = []
  tableParametersModel: TableParametersModel
 
  displayedColumns: string[] = ['text', 'value'];
  isFilled: boolean = false;
  form: FormGroup;
  training :string;
  task:string;
  selectedValue: string;
  selectedTask: string
  allStepParamsRequest: AllStepParamsRequest = {
    parametro_peticion: 'all_steps',
    id_task: ''
  }

  constructor(
    private formBuilder: FormBuilder,
    private router: Router,
    private sharingService: SharingdataService,
    private synService: SynApiService,
    private changeDetectorRefs: ChangeDetectorRef
  ) {
    this.form = this.formBuilder.group({
      treatment: ['']
    });
  }

  ngOnInit(): void {
 
  }


  activateButton() { this.isFilled = !this.isFilled ? true : true }
  /**
   * Devuelve todas las tareas para el tratamiento seleccionado
   * @param treatment Identificador del tratamiento seleccionado
   */
  async selectTreatment(treatment: string) {
    this.selectedTask = ''
    this.selectedValue = treatment
    this.training= treatment
    this.task = ''
    console.log(this.task.length)
    await this.synService.allTasks(this.getQueryParam(this.selectedValue)).subscribe(
      (result) => {
        this.listaTask = result.result

      },
      (error) => { }
    )

    this.isFilled = false

  }
/**
 * MÃ©todo que redirecciona a la ruta seleccionada
 * @param url Ruta seleccionada
 */
  changeComponent(url: string) {

    this.indexToTourModel = new IndexToTourModel(this.getQueryParam(this.selectedValue), 'openoffice')

    this.sharingService.setData(this.indexToTourModel);
    this.sharingService.setAllStepParamsRequest(this.allStepParamsRequest)

    this.router.navigate([url]);//redirects url to new component
  }
  /**
   * Devuelve el parÃ¡metro para las peticiones del tratamiento seleccionado
   * @param treatment Tratamiento seleccionado
   */
  getQueryParam(treatment: string) {
    let result = ''
    switch (treatment) {
      case 'CalificaciÃ³n de incidencias':
        result = 'classification'
        break;
      case 'DetecciÃ³n de incidencias duplicadas':
        result = 'duplicates'
        break;
      case 'RecuperaciÃ³n de incidencias similares':
        result = 'retrieval'
        break;
      case 'AsignaciÃ³n de incidencias':
        result = 'assignation'
        break;
      default:
        break;
    }
    return result
  }
/**
 * Carga la tabla de parÃ¡metros con los datos correspondientes a la tarea seleccionada
 */
  loadTable() {
    this.allStepParamsRequest.id_task = this.selectedTask
    this.task = this.listaTask.find(element => element.task_id.toString() == this.selectedTask).task_description
    this.synService.getAllParams(this.allStepParamsRequest).subscribe(
      (result) => {
        this.tableParametersModel = new TableParametersModel(result.result, 'all_steps')
        
        console.log(this.tableParametersModel)

      },
      (error) => {

      }
    )

  }
  refresh(){
    this.changeDetectorRefs.detectChanges();
  }
  
}
