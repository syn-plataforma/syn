import { Component, Input, OnInit } from '@angular/core';
import { FormBuilder, FormGroup } from '@angular/forms';
import { BehaviorSubject, Observable } from 'rxjs';
import { delay } from 'rxjs/operators';
const ELEMENT_DATA = ["Texto de prueba1","Texto de prueba2","Texto de prueba3","Texto de prueba4"];
interface DatoEjemplo{
    product : string;
    description: string;
    bugSeverity: string;
    priority: string;
    component:string;
}
@Component({
  selector: 'app-display',
  templateUrl: './display.component.html',
  styleUrls: ['./display.component.scss']
})
export class DisplayComponent implements OnInit {
  displayedColumns: string[] = ['position'];
  dataSource = ELEMENT_DATA;
  datoEjemplo : DatoEjemplo= {
    product : "Calc",
    description: "I need to see if this works sorry",
    bugSeverity: "trivial",
    priority: "P4",
    component:"code"

  }
  @Input() data;
  step = 'all_steps';
  frmStepTwo: FormGroup;
  frmStepTwo$: Observable<FormGroup>;
  tituloTratamiento:string
  desarrolladores = [
    "michael.keppler@gmx.de",
    "kalyan_prasad@in.ibm.com",
    "umair_sair@hotmail.com",
    "sergei.kovalchuk@arsysop.ru",
    "edscadding@secondfiddle.org.uk"

  ]
  private myFrmStepTwo$ = new BehaviorSubject<FormGroup>(null);
  myFrmStepTwoListener$: Observable<FormGroup> = this.myFrmStepTwo$.asObservable();
  
  constructor(private formBuilder: FormBuilder) {
    this.frmStepTwo = this.formBuilder.group({
      address: ['']
    })
  }

  ngOnInit(): void {
    this.myFrmStepTwo(this.frmStepTwo);
    console.log(this.tituloTratamiento)
    this.tituloTratamiento = this.data.treatment
    console.log(this.tituloTratamiento)
    this.frmStepTwo$ = this.myFrmStepTwoListener$.pipe(
      delay(0)
    );
  }
  /**
   * Cambia de componente
   * @param form Formulario 
   */
  myFrmStepTwo(form: FormGroup) {
    this.myFrmStepTwo$.next(form);
  }
}
