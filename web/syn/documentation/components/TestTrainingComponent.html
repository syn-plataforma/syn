import { Component, OnInit, ViewChild } from '@angular/core';
import { MatSnackBar } from '@angular/material/snack-bar';
import { Router } from '@angular/router';
import { IndexToTourModel } from 'src/app/core/models/indexToTour.model';
import { InfoRowModel } from 'src/app/core/models/infoRow.model';
import { SharingdataService } from 'src/app/core/services/sharingdata.service';
import { DisplayComponent } from './presenters/display/display.component';
import { InputTestComponent } from './presenters/input-test/input-test.component';
/**
 * Componente que se encargar de mostrar las vistas del "modo directo".
 *
 * Contenido por {@Link AppComponent}
 *
 */
@Component({
  selector: 'app-test-training',
  templateUrl: './test-training.component.html',
  styleUrls: ['./test-training.component.scss']
})
export class TestTrainingComponent implements OnInit {
  isEditable = false;
  dataFromIndex: any;


  @ViewChild('stepOne') stepOneComponent: InputTestComponent;
  @ViewChild('stepTwo') stepTwoComponent: DisplayComponent;
  dataToInput: IndexToTourModel;

  table: InfoRowModel[] = new Array();
  public tituloTratamiento: string
  constructor(private router: Router,
    private sharingService: SharingdataService,private _snackBar: MatSnackBar) {

    this.table.push(new InfoRowModel('name/id', 'name_text', 'value_text'))

    this.sharingService.setTable(this.table)

  }

  ngOnInit(): void {
    if (this.sharingService.getData() === "") {
      this.router.navigate([''])
      this.openSnackBar()
    }else {
      this.dataToInput = this.sharingService.getData()
      console.log(this.dataToInput)
    }
    this.tituloTratamiento = this.dataToInput.treatment
  }
/**
 * Al cerrar el componente limpia los datos compartidos
 */
  ngOnDestroy() {
    this.sharingService.setData("")
    this.sharingService.setTable(new Array())
  }
  
  get frmStepOne() {
    return this.stepOneComponent ? this.stepOneComponent.frmStepOne$ : null;
  }
  get frmStepTwo() {
    return this.stepTwoComponent ? this.stepTwoComponent.frmStepTwo$ : null;
  }

  openSnackBar() {
    this._snackBar.open("Debe de elegir un tratamiento antes de continuar", "Cerrar", {
      duration: 2000,
    });
  }

  }
