/**
 * Componente que se encargar de mostrar las vistas del modo guiado del api.
 *
 * Se encuentra contenido dentro de {@Link AppComponent}
 *
 * Las vistas contenidas son:
 * 1. {@link InputComponent}
 * 2. {@link NlpComponent}
 * 3. {@link EmbeddingComponent}
 * 4. {@link CodebookComponent}
 * 5. {@link VectorizerComponent}
 */

@Component({
  selector: 'app-tour',
  templateUrl: './tour.component.html',
  styleUrls: ['./tour.component.scss']
})
export class TourComponent implements OnInit {

  isEditable = false;
  dataFromIndex: any;

  @ViewChild('stepOne') stepOneComponent: InputComponent;
  @ViewChild('stepTwo') stepTwoComponent: NlpComponent;
  @ViewChild('stepThree') stepThreeComponent: EmbeddingComponent;
  @ViewChild('stepFour') stepFourComponent: CodebookComponent;
  @ViewChild('stepFive') stepFiveComponent: VectorizerComponent;
  @ViewChild('stepSix') stepSixComponent: FinalStepComponent;

  dataToInput: IndexToTourModel;

  allStepParamsRequest: AllStepParamsRequest;

  table: InfoRowModel[] = new Array();

  dataToNlp = [// 'esto es una pruebaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaa' +
    // 'aaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaa' +
    // 'aaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaa' +
    'aaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaa' +
    ' aaaaaaaaaaaaaaaaaaaaaaaaa', ['y esto tmb', 'y esto 2']];

  dataToCodebooks = [[[8, 8, 8, 8, 8, 8], [7, 7, 7, 7, 7, 7, 7, 7]], [8, 7]];
  public tituloTratamiento:string
  constructor(private formBuilder: FormBuilder, private router: Router,
    private sharingService: SharingdataService, private _snackBar: MatSnackBar) {
      this.table.push(new InfoRowModel ('name/id', 'name_text', 'value_text'))
      
      this.sharingService.setTable(this.table)

      console.log(this.sharingService.getTable())
     }

  ngOnInit() {

    console.log(this.sharingService.getData())
    if (this.sharingService.getData() === "") {
      this.router.navigate([''])
      this.openSnackBar()
    }else {
      this.dataToInput = this.sharingService.getData()
      this.allStepParamsRequest = this.sharingService.getAllStepParamsRequest()
      console.log(this.allStepParamsRequest)
    }
    this.tituloTratamiento = this.dataToInput.treatment
  }

  ngOnDestroy() {
    this.sharingService.setData("")
    this.sharingService.setTable(new Array())
  }



  get frmStepOne() {
    return this.stepOneComponent ? this.stepOneComponent.frmStepOne$ : null;
  }

  get frmStepTwo() {
    return this.stepTwoComponent ? this.stepTwoComponent.frmStepTwo$ : null;
  }
  get frmStepThree() {
    return this.stepThreeComponent ? this.stepThreeComponent.frmStepThree$ : null;
  }

  get frmStepFour() {
    return this.stepFourComponent ? this.stepFourComponent.frmStepFour$ : null;
  }

  get frmStepFive() {
    return this.stepFiveComponent ? this.stepFiveComponent.frmStepFive$ : null;
  }
  get frmStepSix() {
    return this.stepSixComponent ? this.stepSixComponent.frmStepSix$ : null;
  }


  openSnackBar() {
    this._snackBar.open("Debe de elegir un tratamiento antes de continuar", "Cerrar", {
      duration: 2000,
    });
  }


  printData() {
    console.log(this.stepOneComponent.devolverDato());
  }

}
